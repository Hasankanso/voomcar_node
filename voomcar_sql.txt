USE voomcar;
SET foreign_key_checks = 0;
CREATE TABLE IF NOT EXISTS alert (`leavingDate` DATETIME ,
`price` DOUBLE ,
`objectId` BINARY(16) NOT NULL PRIMARY KEY ,
`ownerId` BINARY(16) ,  FOREIGN KEY (`ownerId`) REFERENCES `users`(`objectId`),
`created` DATETIME ,
`updated` DATETIME ,
`from` BINARY(16) , FOREIGN KEY (`from`) REFERENCES `location`(`objectId`),
`passenger` BINARY(16) , FOREIGN KEY (`passenger`) REFERENCES `person`(`objectId`),
`to` BINARY(16) , FOREIGN KEY (`to`) REFERENCES `location`(`objectId`));

CREATE TABLE IF NOT EXISTS blConfigurationItemDescription (`defaultValue` VARCHAR(250) ,
`displayName` VARCHAR(250) ,
`hint` VARCHAR(250) ,
`localServiceVersion` VARCHAR(250) ,
`name` VARCHAR(250) ,
`options` VARCHAR(250) ,
`order` DOUBLE ,
`required` BOOLEAN ,
`type` VARCHAR(250) ,
`objectId` BINARY(16) NOT NULL PRIMARY KEY ,
`ownerId` BINARY(16) ,  FOREIGN KEY (`ownerId`) REFERENCES `users`(`objectId`),
`created` DATETIME ,
`updated` DATETIME );

CREATE TABLE IF NOT EXISTS car (`brand` VARCHAR(500) ,
`color` INT ,
`maxLuggage` INT ,
`maxSeats` INT ,
`name` TEXT ,
`picture` VARCHAR(256) ,
`type` INT ,
`user` VARCHAR(250) ,
`year` INT ,
`objectId` BINARY(16) NOT NULL PRIMARY KEY ,
`ownerId` BINARY(16) ,  FOREIGN KEY (`ownerId`) REFERENCES `users`(`objectId`),
`created` DATETIME ,
`updated` DATETIME ,
`driver` BINARY(16) , FOREIGN KEY (`driver`) REFERENCES `driver`(`objectId`));

CREATE TABLE IF NOT EXISTS countryInformation (`code` VARCHAR(250) ,
`countryComponent` VARCHAR(250) ,
`digits` INT ,
`drivingAge` INT ,
`maxPrice` DOUBLE ,
`minPrice` DOUBLE ,
`name` VARCHAR(500) ,
`priceStep` DOUBLE ,
`rateStart` INT ,
`unit` VARCHAR(500) ,
 INDEX (`unit`),
`objectId` BINARY(16) NOT NULL PRIMARY KEY ,
`ownerId` BINARY(16) ,  FOREIGN KEY (`ownerId`) REFERENCES `users`(`objectId`),
`created` DATETIME ,
`updated` DATETIME );

CREATE TABLE IF NOT EXISTS deviceRegistration (`channelName` VARCHAR(45)  NOT NULL,
`deviceId` VARCHAR(45)  NOT NULL,
`deviceToken` VARCHAR(256)  NOT NULL,
`expiration` DATETIME ,
`operatingSystemName` VARCHAR(15)  NOT NULL,
`operatingSystemVersion` VARCHAR(45) ,
`objectId` BINARY(16) NOT NULL PRIMARY KEY ,
`ownerId` BINARY(16) ,  FOREIGN KEY (`ownerId`) REFERENCES `users`(`objectId`),
`created` DATETIME ,
`updated` DATETIME ,
`user` BINARY(16) , FOREIGN KEY (`user`) REFERENCES `Users`(`objectId`));

CREATE TABLE IF NOT EXISTS driver (`objectId` BINARY(16) NOT NULL PRIMARY KEY ,
`ownerId` BINARY(16) ,  FOREIGN KEY (`ownerId`) REFERENCES `users`(`objectId`),
`created` DATETIME ,
`updated` DATETIME ,
`person` BINARY(16) , FOREIGN KEY (`person`) REFERENCES `person`(`objectId`),
`region1` BINARY(16) , FOREIGN KEY (`region1`) REFERENCES `location`(`objectId`),
`region2` BINARY(16) , FOREIGN KEY (`region2`) REFERENCES `location`(`objectId`),
`region3` BINARY(16) , FOREIGN KEY (`region3`) REFERENCES `location`(`objectId`));

CREATE TABLE IF NOT EXISTS location (`name` VARCHAR(500) ,
`placeId` VARCHAR(250)  UNIQUE,
 INDEX (`placeId`),
`position` GEOMETRY ,
`objectId` BINARY(16) NOT NULL PRIMARY KEY ,
`ownerId` BINARY(16) ,  FOREIGN KEY (`ownerId`) REFERENCES `users`(`objectId`),
`created` DATETIME ,
`updated` DATETIME );

CREATE TABLE IF NOT EXISTS Loggers (`level` VARCHAR(250)  NOT NULL,
`logFormat` VARCHAR(250)  NOT NULL,
`name` VARCHAR(250)  NOT NULL,
`policy` VARCHAR(250)  NOT NULL,
`objectId` BINARY(16) NOT NULL PRIMARY KEY ,
`ownerId` BINARY(16) ,  FOREIGN KEY (`ownerId`) REFERENCES `users`(`objectId`),
`created` DATETIME ,
`updated` DATETIME );

CREATE TABLE IF NOT EXISTS person (`bio` VARCHAR(500) ,
`birthday` DATETIME ,
`chattiness` INT ,
`firstName` VARCHAR(250) ,
`gender` BOOLEAN ,
`image` VARCHAR(256) ,
`lastName` VARCHAR(500) ,
`token` VARCHAR(500) ,
`objectId` BINARY(16) NOT NULL PRIMARY KEY ,
`ownerId` BINARY(16) ,  FOREIGN KEY (`ownerId`) REFERENCES `users`(`objectId`),
`created` DATETIME ,
`updated` DATETIME ,
`countryInformations` BINARY(16) , FOREIGN KEY (`countryInformations`) REFERENCES `countryInformation`(`objectId`),
`statistics` BINARY(16)  UNIQUE, FOREIGN KEY (`statistics`) REFERENCES `userStatistics`(`objectId`));

CREATE TABLE IF NOT EXISTS rate (`comment` VARCHAR(500) ,
`grade` DOUBLE  NOT NULL,
`reason` INT ,
`objectId` BINARY(16) NOT NULL PRIMARY KEY ,
`ownerId` BINARY(16) ,  FOREIGN KEY (`ownerId`) REFERENCES `users`(`objectId`),
`created` DATETIME ,
`updated` DATETIME ,
`rater` BINARY(16) , FOREIGN KEY (`rater`) REFERENCES `person`(`objectId`),
`ride` BINARY(16) , FOREIGN KEY (`ride`) REFERENCES `ride`(`objectId`),
`target` BINARY(16) , FOREIGN KEY (`target`) REFERENCES `person`(`objectId`));

CREATE TABLE IF NOT EXISTS reserve (`luggages` INT ,
`seats` INT ,
`status` VARCHAR(250) ,
`objectId` BINARY(16) NOT NULL PRIMARY KEY ,
`ownerId` BINARY(16) ,  FOREIGN KEY (`ownerId`) REFERENCES `users`(`objectId`),
`created` DATETIME ,
`updated` DATETIME ,
`person` BINARY(16) , FOREIGN KEY (`person`) REFERENCES `person`(`objectId`),
`ride` BINARY(16) , FOREIGN KEY (`ride`) REFERENCES `ride`(`objectId`));

CREATE TABLE IF NOT EXISTS ride (`acAllowed` BOOLEAN ,
`availableLuggages` INT ,
`availableSeats` INT ,
`comment` VARCHAR(400) ,
`kidSeat` BOOLEAN ,
`leavingDate` DATETIME ,
`map` VARCHAR(256) ,
`maxLuggages` INT ,
`maxSeats` INT ,
`musicAllowed` BOOLEAN ,
`petsAllowed` BOOLEAN ,
`price` DOUBLE ,
`smokingAllowed` BOOLEAN ,
`status` VARCHAR(250) ,
`stopTime` INT ,
`objectId` BINARY(16) NOT NULL PRIMARY KEY ,
`ownerId` BINARY(16) ,  FOREIGN KEY (`ownerId`) REFERENCES `users`(`objectId`),
`created` DATETIME ,
`updated` DATETIME ,
`car` BINARY(16) , FOREIGN KEY (`car`) REFERENCES `car`(`objectId`),
`driver` BINARY(16) , FOREIGN KEY (`driver`) REFERENCES `driver`(`objectId`),
`from` BINARY(16) , FOREIGN KEY (`from`) REFERENCES `location`(`objectId`),
`to` BINARY(16) , FOREIGN KEY (`to`) REFERENCES `location`(`objectId`));

CREATE TABLE IF NOT EXISTS schedule (`endDate` DATETIME ,
`friday` BOOLEAN ,
`monday` BOOLEAN ,
`saturday` BOOLEAN ,
`startDate` DATETIME ,
`sunday` BOOLEAN ,
`thursday` BOOLEAN ,
`tuesday` BOOLEAN ,
`wednesday` BOOLEAN ,
`objectId` BINARY(16) NOT NULL PRIMARY KEY ,
`ownerId` BINARY(16) ,  FOREIGN KEY (`ownerId`) REFERENCES `users`(`objectId`),
`created` DATETIME ,
`updated` DATETIME );

CREATE TABLE IF NOT EXISTS scheduleRide (`objectId` BINARY(16) NOT NULL PRIMARY KEY ,
`ownerId` BINARY(16) ,  FOREIGN KEY (`ownerId`) REFERENCES `users`(`objectId`),
`created` DATETIME ,
`updated` DATETIME ,
`driver` BINARY(16) , FOREIGN KEY (`driver`) REFERENCES `driver`(`objectId`),
`schedule` BINARY(16) , FOREIGN KEY (`schedule`) REFERENCES `schedule`(`objectId`));

CREATE TABLE IF NOT EXISTS users (`blUserLocale` VARCHAR(250)  NOT NULL,
`email` VARCHAR(250)  NOT NULL,
 INDEX (`email`),
`oAuthIdentities` JSON ,
`password` VARCHAR(250)  NOT NULL,
`phone` VARCHAR(250)  NOT NULL UNIQUE,
 INDEX (`phone`),
`verificationCode` TEXT ,
`objectId` BINARY(16) NOT NULL PRIMARY KEY ,
`ownerId` BINARY(16) ,  FOREIGN KEY (`ownerId`) REFERENCES `users`(`objectId`),
`created` DATETIME ,
`updated` DATETIME ,
`person` BINARY(16) , FOREIGN KEY (`person`) REFERENCES `person`(`objectId`),
`bckls__userStatusId` INT ,
`bckls__lastLogin` DATETIME );

CREATE TABLE IF NOT EXISTS userStatistics (`acomplishedRides` INT  NOT NULL,
`canceledRides` INT  NOT NULL,
`fives` INT ,
`fours` INT ,
`ones` INT ,
`rateAverage` DOUBLE ,
`ratesCount` INT  NOT NULL,
`threes` INT ,
`twos` INT ,
`objectId` BINARY(16) NOT NULL PRIMARY KEY ,
`ownerId` BINARY(16) ,  FOREIGN KEY (`ownerId`) REFERENCES `users`(`objectId`),
`created` DATETIME ,
`updated` DATETIME );
